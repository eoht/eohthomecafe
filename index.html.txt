<script>
  // ── STATE ──
  const state = {
    type: null,
    alcohol: null,
    mood: null,
    ingredients: new Set()
  };

  let currentStep = 1;
  let currentTab = 'dairy';

  // ── INGREDIENT DATABASE ──
  const ingrDB = {
    dairy: ['Whole milk','Oat milk','Almond milk','Soy milk','Coconut milk','Heavy cream','Half & half','Condensed milk','Evaporated milk'],
    coffee: ['Espresso','Instant coffee','Drip coffee','Coffee grounds','Cold brew'],
    tea: ['Black tea','Green tea','Matcha powder','Oolong tea','Jasmine tea','Earl Grey','Chamomile','Brown sugar pearls','Taro powder'],
    sweetener: ['White sugar','Brown sugar','Honey','Maple syrup','Agave','Vanilla syrup','Caramel syrup','Simple syrup','Stevia'],
    spice: ['Cinnamon','Vanilla extract','Cardamom','Nutmeg','Ginger','Cocoa powder','Chocolate syrup','Peppermint'],
    spirits: ['Vodka','Rum','Gin','Tequila','Whiskey','Baileys','Kahlúa','Amaretto','Triple sec','Champagne/Prosecco'],
    fruit: ['Lemon','Lime','Orange','Strawberry','Mango','Passion fruit','Raspberry','Pineapple','Coconut water'],
    other: ['Ice','Whipped cream','Sea salt','Boba/tapioca pearls','Rose water','Lavender syrup','Sparkling water','Club soda']
  };

  // ── RENDER CHIPS ──
  function renderChips(cat) {
    const container = document.getElementById('ingrChips');
    container.innerHTML = ingrDB[cat].map(item => {
      const picked = state.ingredients.has(item);
      return `<button class="ingr-chip ${picked ? 'picked' : ''}" onclick="toggleIngr(this, '${item}')">${item}</button>`;
    }).join('');
  }

  function switchTab(el) {
    document.querySelectorAll('.ingr-tab').forEach(t => t.classList.remove('active'));
    el.classList.add('active');
    currentTab = el.dataset.cat;
    renderChips(currentTab);
  }

  function toggleIngr(el, item) {
    if (state.ingredients.has(item)) {
      state.ingredients.delete(item);
      el.classList.remove('picked');
    } else {
      state.ingredients.add(item);
      el.classList.add('picked');
    }
    updateSelectedDisplay();
  }

  function removeIngr(item) {
    state.ingredients.delete(item);
    // also un-pick chip if visible
    document.querySelectorAll('.ingr-chip').forEach(c => {
      if (c.textContent.replace(' ✓','').trim() === item) c.classList.remove('picked');
    });
    updateSelectedDisplay();
  }

  function addCustomIngr() {
    const input = document.getElementById('customInput');
    const val = input.value.trim();
    if (!val) return;
    state.ingredients.add(val);
    input.value = '';
    updateSelectedDisplay();
  }

  function updateSelectedDisplay() {
    const wrap = document.getElementById('selectedTags');
    const noText = document.getElementById('noIngrText');
    const nextBtn = document.getElementById('step4Next');
    if (state.ingredients.size === 0) {
      wrap.innerHTML = '<span class="no-ingr" id="noIngrText">Tap items below to add them ↓</span>';
      nextBtn.disabled = true;
    } else {
      wrap.innerHTML = [...state.ingredients].map(item =>
        `<div class="sel-tag">${item}<button onclick="removeIngr('${item}')">×</button></div>`
      ).join('');
      nextBtn.disabled = false;
    }
    // update step summary
    if (state.ingredients.size > 0) {
      const preview = [...state.ingredients].slice(0,3).join(', ');
      const extra = state.ingredients.size > 3 ? ` +${state.ingredients.size - 3} more` : '';
      document.getElementById('ss4').textContent = preview + extra;
    }
  }

  // ── STEP NAVIGATION ──
  function pickType(btn) {
    document.querySelectorAll('#step1 .opt-btn').forEach(b => b.classList.remove('selected'));
    btn.classList.add('selected');
    state.type = btn.dataset.val;
    const labels = { coffee:'☕ Coffee Latte', matcha:'🍵 Matcha Latte', tea:'🫖 Tea Latte', cocktail:'🍹 Cocktail', any:'🎲 Surprise me' };
    document.getElementById('ss1').textContent = labels[state.type];
    setTimeout(() => advanceTo(2), 250);
  }

  function pickAlcohol(btn) {
    document.querySelectorAll('#step2 .opt-btn').forEach(b => b.classList.remove('selected'));
    btn.classList.add('selected');
    state.alcohol = btn.dataset.val;
    const labels = { 'non-alcoholic':'🍃 Non-Alcoholic', alcoholic:'🍸 Alcoholic', both:'✨ Both' };
    document.getElementById('ss2').textContent = labels[state.alcohol];
    setTimeout(() => advanceTo(3), 250);
  }

  function pickMood(btn) {
    document.querySelectorAll('#step3 .opt-btn').forEach(b => b.classList.remove('selected'));
    btn.classList.add('selected');
    state.mood = btn.dataset.val;
    const labels = { cozy:'🕯️ Cozy & warm', refreshing:'🌊 Cool & refreshing', indulgent:'🍫 Rich & indulgent', energizing:'⚡ Wake me up!' };
    document.getElementById('ss3').textContent = labels[state.mood];
    setTimeout(() => advanceTo(4), 250);
  }

  function doneIngredients() {
    if (state.ingredients.size === 0) return;
    markDone(4);
    updateProgress(4);
    document.getElementById('generateBtn').style.display = 'block';
    document.getElementById('generateBtn').scrollIntoView({ behavior: 'smooth', block: 'center' });
  }

  function advanceTo(n) {
    markDone(n - 1);
    openStep(n);
    updateProgress(n);
    renderChips(currentTab); // refresh chips when step 4 opens
  }

  function markDone(n) {
    const card = document.getElementById('step' + n);
    if (!card) return;
    card.classList.remove('active');
    card.classList.add('done');
    document.getElementById('sc' + n).style.display = 'none';
    currentStep = n + 1;
  }

  function openStep(n) {
    const card = document.getElementById('step' + n);
    if (!card) return;
    card.classList.remove('done');
    card.classList.add('active');
    document.getElementById('sc' + n).style.display = '';
    card.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
  }

  function toggleStep(n) {
    const card = document.getElementById('step' + n);
    if (card.classList.contains('active')) {
      // collapse (allow re-editing)
    } else {
      card.classList.add('active');
      card.classList.remove('done');
      document.getElementById('sc' + n).style.display = '';
      if (n === 4) renderChips(currentTab);
    }
  }

  function updateProgress(active) {
    for (let i = 1; i <= 4; i++) {
      const dot = document.getElementById('pd' + i);
      if (i < active) { dot.className = 'prog-dot done'; }
      else if (i === active) { dot.className = 'prog-dot active'; }
      else { dot.className = 'prog-dot'; }
    }
  }

  // ── GENERATE ──
  async function generateDrinks() {
    const btn = document.getElementById('generateBtn');
    const results = document.getElementById('resultsSection');
    const list = document.getElementById('drinksList');

    btn.disabled = true;
    btn.textContent = '✦ Brewing your menu…';

    results.classList.add('visible');
    list.innerHTML = `
      <div class="loading-card">
        <div class="loading-steam"><div class="steam-line"></div><div class="steam-line"></div><div class="steam-line"></div></div>
        <div class="loading-cup">☕</div>
        <div class="loading-text">Your barista is crafting<span class="loading-dots"><span></span><span></span><span></span></span></div>
      </div>`;

    results.scrollIntoView({ behavior: 'smooth', block: 'start' });

    const typeMap = {
      any: 'any drink type — choose the best fit from coffee latte, matcha latte, tea latte (Molly Tea / Chagee / Tiger Sugar style), or cocktail',
      coffee: 'coffee lattes and espresso-based drinks only',
      matcha: 'matcha lattes and matcha-based drinks only',
      tea: 'tea lattes in a Molly Tea / Chagee / Tiger Sugar style (brown sugar milk tea, dirty matcha, jasmine latte, oolong, etc.)',
      cocktail: 'cocktails and spirit-based drinks only'
    };

    const alcMap = {
      'non-alcoholic': 'strictly non-alcoholic — no alcohol',
      'alcoholic': 'alcoholic — use spirits where possible',
      'both': 'a mix: include both alcoholic and non-alcoholic drinks across the 3 suggestions'
    };

    const ingredients = [...state.ingredients].join(', ');

    const prompt = `You are Eunice's personal café barista. Create 3 drinks using ONLY the listed ingredients. Be creative and café-quality.

Ingredients available: ${ingredients}
Drink category: ${typeMap[state.type]}
Alcohol: ${alcMap[state.alcohol]}
Mood/vibe: ${state.mood}

Return ONLY valid JSON, no markdown, no code fences:
{"drinks":[{"name":"string","emoji":"string","type":"coffee|matcha|tea|cocktail","isAlcoholic":false,"description":"2 evocative sentences","ingredients":["amount + item"],"steps":["clear step"]}]}

Rules:
- Use ONLY provided ingredients. Assume ice, water, a cup and spoon are available.
- Match the ${state.mood} vibe in the name and description
- Names should be café-worthy and poetic, not generic (e.g. "Velvet Brown Sugar Latte" not "Brown Sugar Latte")
- 3–5 steps per drink, each 1–2 sentences
- Include amounts in ingredients (e.g. "1 shot espresso", "150ml oat milk")
- Make drinks distinct from each other
- Tea lattes: think Molly Tea, Chagee, Tiger Sugar — brown sugar boba, dirty matchas, iced jasmine lattes, etc.`;

    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          messages: [{ role: "user", content: prompt }]
        })
      });
      const data = await res.json();
      const raw = data.content.map(c => c.text || '').join('');
      const clean = raw.replace(/```json|```/g, '').trim();
      const parsed = JSON.parse(clean);
      renderDrinks(parsed.drinks);
      document.getElementById('resultsMeta').textContent = parsed.drinks.length + ' drinks for you';
    } catch(e) {
      list.innerHTML = `<div class="error-card">Oops! Something went wrong. Try checking your ingredients or give it another shot ☕</div>`;
    }

    btn.disabled = false;
    btn.textContent = '✦ Craft Eunice\'s Menu ✦';
  }

  function renderDrinks(drinks) {
    const list = document.getElementById('drinksList');
    const typeClass = { coffee:'badge-coffee', matcha:'badge-matcha', tea:'badge-tea', cocktail:'badge-cocktail' };
    list.innerHTML = drinks.map((d, i) => `
      <div class="drink-card">
        <div class="dc-top">
          <div class="dc-emoji">${d.emoji}</div>
          <div class="dc-info">
            <div class="dc-name">${d.name}</div>
            <div class="dc-badges">
              <span class="badge ${d.isAlcoholic ? 'badge-alc' : 'badge-nonalc'}">${d.isAlcoholic ? '🍸 Alcoholic' : '🍃 Non-Alc'}</span>
              <span class="badge ${typeClass[d.type] || 'badge-coffee'}">${cap(d.type)}</span>
            </div>
          </div>
        </div>
        <div class="dc-body">
          <div class="dc-desc">${d.description}</div>
          <div class="recipe-toggle" onclick="toggleRecipe(this)">
            <div class="recipe-toggle-left">📋 View Recipe</div>
            <span class="rt-arrow">▼</span>
          </div>
          <div class="recipe-body">
            <div class="rb-section">Ingredients</div>

            </div>
            <div class="rb-section">Steps</div>
            <div class="rb-steps">

            </div>
          </div>
        </div>
      </div>
    `).join('');
  }

  function toggleRecipe(el) {
    el.classList.toggle('open');
    el.nextElementSibling.classList.toggle('open');
  }

  function cap(s) { return s ? s[0].toUpperCase() + s.slice(1) : ''; }

  function resetAll() {
    state.type = null; state.alcohol = null; state.mood = null;
    state.ingredients.clear();
    location.reload();
  }

  // Init chips
  renderChips('dairy');
